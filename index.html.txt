// ══════════════════════════════════════════
//  STATE
// ══════════════════════════════════════════
let apiKey = '';
let isListening = false;
let recognition = null;
let synth = window.speechSynthesis;
let conversationHistory = [];
let isSpeaking = false;

// ══════════════════════════════════════════
//  INIT
// ══════════════════════════════════════════
window.addEventListener('load', () => {
  apiKey = localStorage.getItem('nexus_api_key') || '';
  if (apiKey) {
    document.getElementById('apiModal').classList.add('hidden');
    initRecognition();
  }
  // Load history
  const saved = localStorage.getItem('nexus_history');
  if (saved) conversationHistory = JSON.parse(saved);
});

// ══════════════════════════════════════════
//  API KEY
// ══════════════════════════════════════════
function saveApiKey() {
  const val = document.getElementById('apiKeyInput').value.trim();
  if (!val.startsWith('sk-ant')) {
    showToast('⚠ Clave inválida');
    return;
  }
  apiKey = val;
  localStorage.setItem('nexus_api_key', apiKey);
  document.getElementById('apiModal').classList.add('hidden');
  initRecognition();
  showToast('✓ NEXUS ACTIVADO');
}

function clearKey() {
  if (!confirm('¿Cambiar API key?')) return;
  localStorage.removeItem('nexus_api_key');
  apiKey = '';
  document.getElementById('apiKeyInput').value = '';
  document.getElementById('apiModal').classList.remove('hidden');
}

// ══════════════════════════════════════════
//  SPEECH RECOGNITION
// ══════════════════════════════════════════
function initRecognition() {
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  if (!SpeechRecognition) {
    showToast('⚠ Usa Safari en iPhone');
    return;
  }
  recognition = new SpeechRecognition();
  recognition.lang = 'es-ES';
  recognition.continuous = false;
  recognition.interimResults = true;
  recognition.maxAlternatives = 1;

  recognition.onstart = () => {
    setOrbState('listening');
    setStatus('ESCUCHANDO...');
    document.getElementById('soundBars').classList.add('active');
  };

  recognition.onresult = (e) => {
    let interim = '';
    let final = '';
    for (let i = e.resultIndex; i < e.results.length; i++) {
      const t = e.results[i][0].transcript;
      if (e.results[i].isFinal) final += t;
      else interim += t;
    }
    const display = final || interim;
    document.getElementById('transcriptArea').textContent = display;
    if (final) processQuery(final);
  };

  recognition.onerror = (e) => {
    setOrbState('error');
    setStatus('ERROR: ' + e.error.toUpperCase());
    isListening = false;
    document.getElementById('soundBars').classList.remove('active');
    setTimeout(() => { setOrbState('idle'); setStatus('LISTO'); }, 2000);
  };

  recognition.onend = () => {
    isListening = false;
    document.getElementById('soundBars').classList.remove('active');
    if (document.getElementById('mainOrb').className.includes('listening')) {
      setOrbState('idle');
      setStatus('LISTO');
    }
  };
}

function toggleListen() {
  if (!apiKey) { showToast('Configura tu API key'); return; }
  if (!recognition) { showToast('⚠ Usa Safari en iPhone'); return; }
  if (isSpeaking) { stopSpeaking(); return; }
  if (isListening) {
    recognition.stop();
    isListening = false;
    setOrbState('idle'); setStatus('LISTO');
  } else {
    try {
      recognition.start();
      isListening = true;
    } catch(e) {
      recognition.stop();
      setTimeout(() => { recognition.start(); isListening = true; }, 300);
    }
  }
}

const SYSTEM_PROMPT = `Eres NEXUS, un agente de IA avanzado para iPhone. Eres el asistente personal de Joan Carles, también eres su copiloto en su proyecto de trading llamado NEXUS.

Eres conciso, inteligente, y directo. Hablas en español. Tus respuestas son cortas y claras (máx 2-3 frases para voz) a menos que te pidan más detalle.

Tienes capacidades de acción. Cuando el usuario pida algo que requiera abrir una app o hacer una acción, incluye al final de tu respuesta una línea JSON especial:
ACTION:{"type":"maps","query":"destino"} - para Google Maps
ACTION:{"type":"call","number":"número"} - para llamar
ACTION:{"type":"whatsapp","number":"número","message":"mensaje"} - para WhatsApp
ACTION:{"type":"web","url":"url"} - para abrir web
ACTION:{"type":"email","to":"email","subject":"asunto","body":"cuerpo"} - para email
ACTION:{"type":"sms","number":"número","message":"mensaje"} - para SMS

Ejemplos:
- "Lleva a Joan Carles a la Sagrada Familia" → responde + ACTION:{"type":"maps","query":"Sagrada Familia Barcelona"}
- "Llama a mamá" → responde + ACTION:{"type":"call","number":"mamá"}
- "Manda whatsapp a Pedro que llego tarde" → ACTION:{"type":"whatsapp","number":"Pedro","message":"Llego tarde"}

Para consultas de NEXUS trading, responde con análisis y consejos basados en tu conocimiento de la plataforma.
Para el tiempo, responde con lo que sabes y sugiere abrir Weather.
Para recordatorios, confirma y sugiere añadirlo.

Siempre sé útil, proactivo y natural.`;

async function processQuery(query) {
  setOrbState('thinking');
  setStatus('PROCESANDO...');
  document.getElementById('responseArea').classList.remove('visible');
  document.getElementById('actionButtons').classList.remove('visible');

  conversationHistory.push({ role: 'user', content: query });

  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': apiKey,
        'anthropic-version': '2023-06-01',
        'anthropic-dangerous-direct-browser-access': 'true'
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 500,
        system: SYSTEM_PROMPT,
        messages: conversationHistory.slice(-10)
      })
    });

    if (!res.ok) {
      const err = await res.json();
      throw new Error(err.error?.message || 'API error ' + res.status);
    }

    const data = await res.json();
    const fullText = data.content[0].text;

    // Parse action
    const actionMatch = fullText.match(/ACTION:(\{.*\})/);
    let action = null;
    let displayText = fullText;

    if (actionMatch) {
      try { action = JSON.parse(actionMatch[1]); } catch(e) {}
      displayText = fullText.replace(/ACTION:\{.*\}/, '').trim();
    }

    conversationHistory.push({ role: 'assistant', content: fullText });
    saveHistory();

    showResponse(displayText, action);
    speak(displayText);

  } catch(e) {
    setOrbState('error');
    setStatus('ERROR');
    showResponse('⚠ ' + e.message, null);
    setTimeout(() => { setOrbState('idle'); setStatus('LISTO'); }, 3000);
  }
}

function showResponse(text, action) {
  const area = document.getElementById('responseArea');
  area.innerHTML = '';
  if (action) {
    const tag = document.createElement('div');
    tag.className = 'action-tag';
    tag.textContent = '⚡ ' + action.type.toUpperCase();
    area.appendChild(tag);
    area.appendChild(document.createElement('br'));
  }
  const p = document.createElement('span');
  p.textContent = text;
  area.appendChild(p);
  area.classList.add('visible');

  if (action) showActionButtons(action);
}

function showActionButtons(action) {
  const container = document.getElementById('actionButtons');
  container.innerHTML = '';

  let buttons = [];

  if (action.type === 'maps') {
    buttons.push({ icon: '🗺️', label: 'ABRIR MAPS', fn: () => {
      const url = `https://maps.google.com/?q=${encodeURIComponent(action.query)}`;
      window.open(url, '_blank');
    }});
  } else if (action.type === 'call') {
    buttons.push({ icon: '📞', label: 'LLAMAR', fn: () => {
      window.open(`tel:${action.number}`);
    }});
  } else if (action.type === 'whatsapp') {
    buttons.push({ icon: '💬', label: 'WHATSAPP', fn: () => {
      const num = action.number.replace(/\D/g,'');
      const msg = encodeURIComponent(action.message || '');
      const url = num
        ? `https://wa.me/${num}?text=${msg}`
        : `https://wa.me/?text=${msg}`;
      window.open(url, '_blank');
    }});
  } else if (action.type === 'web') {
    buttons.push({ icon: '🌐', label: 'ABRIR WEB', fn: () => window.open(action.url, '_blank') });
  } else if (action.type === 'email') {
    buttons.push({ icon: '📧', label: 'EMAIL', fn: () => {
      const url = `mailto:${action.to}?subject=${encodeURIComponent(action.subject||'')}&body=${encodeURIComponent(action.body||'')}`;
      window.open(url);
    }});
  } else if (action.type === 'sms') {
    buttons.push({ icon: '💬', label: 'SMS', fn: () => {
      window.open(`sms:${action.number}&body=${encodeURIComponent(action.message||'')}`);
    }});
  }

  // Always add dismiss
  buttons.push({ icon: '✓', label: 'OK', fn: () => {
    container.classList.remove('visible');
    container.innerHTML = '';
  }});

  buttons.forEach(b => {
    const btn = document.createElement('button');
    btn.className = 'action-btn';
    btn.innerHTML = `<span class="btn-icon">${b.icon}</span>${b.label}`;
    btn.onclick = b.fn;
    container.appendChild(btn);
  });

  container.classList.add('visible');
}

function speak(text) {
  if (!synth) return;
  synth.cancel();
  // Strip action JSON and clean
  const clean = text.replace(/ACTION:\{.*\}/g, '').replace(/⚡.*\n?/g,'').trim();
  if (!clean) { setOrbState('idle'); setStatus('LISTO'); return; }

  const utt = new SpeechSynthesisUtterance(clean);
  utt.lang = 'es-ES';
  utt.rate = 1.0;
  utt.pitch = 1.0;

  // Try to find a good Spanish voice
  const voices = synth.getVoices();
  const esVoice = voices.find(v => v.lang.startsWith('es') && v.localService)
    || voices.find(v => v.lang.startsWith('es'));
  if (esVoice) utt.voice = esVoice;

  utt.onstart = () => { isSpeaking = true; setOrbState('speaking'); setStatus('HABLANDO...'); };
  utt.onend = () => { isSpeaking = false; setOrbState('idle'); setStatus('LISTO'); };
  utt.onerror = () => { isSpeaking = false; setOrbState('idle'); setStatus('LISTO'); };

  synth.speak(utt);
}

function stopSpeaking() {
  synth.cancel();
  isSpeaking = false;
  setOrbState('idle');
  setStatus('LISTO');
}

function openHistory() {
  const panel = document.getElementById('historyPanel');
  const list = document.getElementById('historyList');
  list.innerHTML = '';

  const pairs = [];
  for (let i = 0; i < conversationHistory.length - 1; i += 2) {
    if (conversationHistory[i]?.role === 'user') {
      pairs.push({ user: conversationHistory[i].content, agent: conversationHistory[i+1]?.content || '' });
    }
  }

  if (pairs.length === 0) {
    list.innerHTML = '<div style="color:rgba(0,245,255,0.3);font-size:12px;text-align:center;padding:20px">Sin historial aún</div>';
  } else {
    pairs.slice(-15).reverse().forEach(p => {
      const item = document.createElement('div');
      item.className = 'history-item';
      const agentText = p.agent.replace(/ACTION:\{.*\}/g, '').trim();
      item.innerHTML = `<div class="h-user">▶ ${p.user}</div><div class="h-agent">${agentText}</div>`;
      list.appendChild(item);
    });
  }

  panel.classList.add('open');
}

function closeHistory() {
  document.getElementById('historyPanel').classList.remove('open');
}
